//结构体数组的转换
func SymbolTraderInfoToMT4Accounts(ins []*SymbolTraderInfo) []*Mt4Account {
	outs := make([]*Mt4Account, 0)
	for _, in := range ins {
		out := &Mt4Account{
			BrokerID: in.BrokerID,
			Account:  in.Account,
		}
		outs = append(outs, out)
	}
	return outs
}


//双数组使用map进行合并查重
func mergeAccountsScoreToSymbolTraders(ins []*mod.SymbolTraderInfo, accs []*mod.AccountData) []*mod.SymbolTraderInfo {
	if len(accs) == 0 {
		return ins
	}
	scoreMap := make(map[mod.Mt4Account]float64)
	for _, acc := range accs {
		mt4acc := mod.Mt4Account{
			BrokerID: acc.BrokerID,
			Account:  acc.Account,
		}
		scoreMap[mt4acc] = acc.Score
	}

	for _, in := range ins {
		mt4acc := mod.Mt4Account{
			BrokerID: in.BrokerID,
			Account:  in.Account,
		}
		s, ok := scoreMap[mt4acc]
		if ok {
			in.Score = s
		}
	}
	return ins
}
